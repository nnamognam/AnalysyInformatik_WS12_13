\chapter{Mengen}


\section{Definition}

\textbf{1)} \\Eine Menge ist eine Ansammlung verschiedener Objekte. \\ \\
\textbf{2)} \\Die Objekte einer Menge heißen \underline{Elemente}. \\ \\
\textit{Notation:} \\
$a \in M$ heißt: $a$ ist Element der Menge $M$ \\
$a \notin M$ heißt: $a$ ist kein Element der Menge $M$
\textbf{3)} \\ \\Sei $M$ eine Menge. Eine Menge $U$ heißt Teilmenge von $M$, wenn jedes Element von $U$ auch ein Element von $M$ ist. \\ \\
\textit{Notation:} \\
$U \subseteq M$ heißt: $U$ ist Teilmenge der Menge $M$ \\
$U \not \subseteq M$ heißt: $U$ ist keine Teilmenge der Menge $M$


\section{Beispiele}

\textbf{1)} \\Sei $M$ die Menge aller Studierenden in L1, \\
$W$ die Menge aller weiblichen Studierenden in L1, \\
$F$ die Menge aller Frauen. \\ \\
Dann gilt: $W \subseteq M, W \subseteq F, M \not \subseteq F, F \not \subseteq M$. \\ \\
\textbf{2)} \\Die Menge der natürlichen Zahlen: $\mathbb+{N}=\{1,2,3,..\}$ \\ \\
G sei Menge der geraden natürlichen Zahlen. \\
$G := \{n \in \mathbb{N} |$ $n$ gerade $\}$ \\
\hphantom{spaces}$ G := \{2m | m \in \mathbb{N}\}$ \\
\hphantom{spaces}$ N := \{2,4,6,8,...\}$ \\ \\
Es gilt $G \subseteq \mathbb{N}$, $\mathbb{N} \not \subseteq G$. \\ \\
\textbf{3)} \\Die Menge der ganzen Zahlen: \\
$\mathbb{Z} = \{0,-1,1,-2,2,...\}$ \\ \\ \\
\textbf{4)} \\Menge der rationalen Zahlen: \\
$\mathbb{Q} = \frac{a}{b} | a,b \in \mathbb{Z}, b \not = 0\}$ \\ \\
\textbf{5)} \\Die Menge ohne Elemente heißt \underline{leere Menge}. \\
\textit{Symbol:} $\emptyset = \{\}$ \\ \\
\framebox{Bemerkung} \\
\hphantom{spaces} Für jede Menge $M$ gilt: $\emptyset \subseteq M$ \\
\hphantom{spaces} $\mathbb{N} \subseteq \mathbb{Z} \subseteq \mathbb{Q}$


\section{Definition}

Sei $M$ eine Menge, $U,V \subseteq M$ Teilmengen. \\ \\
\textbf{1)} \\Die Vereinigung von $U$ und $V$ ist $U \cup V := \{x \in M | x \in U$ oder $x \in V\}$ \\ \\
\textbf{2)} \\Der Durchschnitt von $U$ und $V$ ist $U \cap V := \{x \in M | x \in U$ oder $x \in V\}$ \\ \\
\textbf{3)} \\Die Differenzmenge von $U$ und $V$ ist $U \slash V := \{x \in U | x \not \in V\}$ \\ \\
\textbf{4)} \\Das Komplement von $U$ ist $U^c = M \slash U = \{x \in M | x \not \in U\}$ \\ \\
\textit{Beispiel} \\
Sei $M = \mathbb{N}$.\\
$\{1,3\} \cup \{3,5\} = \{1,3,5\}$ \\
$\{1,3\} \cap \{3,5\} = \{3\}$ \\
$\{1,3\} \cap \{2,4,7\} = \emptyset$ ,also sind $\{1,3\}$ und $\{2,4,7\}$ disjunkt. \\
$\{1,2,3\} \slash \{3,4,5\} = \{1,2\}$ \\
$\{1,3,5\}^c = \{2,4,6,7,8,...\}$

\section{Prinzip der vollständigen Induktion}
Für jedes $n \in \mathbb{N}$ sei eine Aussage $A(n)$ gegeben. \\
\underline{Ziel:} Beweisen, dass $A(n)$ für jedes $n \in \mathbb{N}$ wahr ist.\\

\textbf{1)} \\Induktionsanfang: $A(1)$ ist wahr. \\ \\
\textbf{2)} \\Induktionsschritt: Wenn für ein $n \in \mathbb{N}$ $A(n)$ wahr ist, dann ist auch $A(n+1)$ wahr.

Elemente von $A_1 \times ... \times A_k$ heißen \underline{k-Tupel}. \\

\section{Definition}

	Sei K angeordneter Körper M$\subseteq$K eine Teilmenge a$\in$K
		\begin{itemize}
			\item[1] M$\leq$a heißt x$\leq$a$\forall$x$\in$M Analog M$\geq$a etc.
			\item[2] a heißt obere Schranke von M, wenn M$\leq$a
			\item[3] M heißt nach oben beschränkt, wenn M eine obere Schranke hat.\\
							 Analog nach unten beschränkt.\\
							 M heißt beschränkt, wenn M nach oben $\wedge$ unten berschränkt ist.
			\item[4] a heißt Maximum von M wenn M$\leq$a und a$\in$M\\
							 Minimum analog.
			\item[Notation:] a=max(M) bzw. a=min(M) 
		\end{itemize}

\subsection{Bemerkung}
	Wenn M ein Maximum hat, ist es eindeutig.

\subsection{Beweis}
	Sei a,b$\in$M, M$\leq$a , M$\leq$b dann\\
	b$\leq$a und a$\leq$b $\Rightarrow$ a=b $_{\square}$

\subsection{Beispiel}
	K=$\mathbb{Q}$
		\begin{itemize}
			\item[1] M=$\mathbb{N}$ Sei a$\in\mathbb{Q}$a$\leq\mathbb{N}$ a$\leq$ n $\forall$n$\in\mathbb{N}$
							 $\mathbb{N}$ ist nach unten beschränkt 1=min($\mathbb{N}$).\\
							 $\mathbb{N}$ ist nicht nach oben beschränkt.
			\item[2] M$=\{\frac{-1}{n}|n\in\mathbb{N}\}0\notin$M\\
							 -1=min(M)$\Rightarrow$ M nach unten beschränkt\\
							 M$<$0 $\Rightarrow$ M ist nach oben beschränkt hat aber kein Maximum
			\item[3] M$=\{\frac{-1}{n}|n\in\mathbb{N}\}\cup\{0\}$ min(M)$=-1|max(M)=0$
			\item[4] M$=\emptyset$ hat kein Maximum und kein Minimum jedes a$\in\mathbb{Q}$ erfüllt a$\leq$ M und M$\leq$a
		\end{itemize}  

\subsection{Beweis zu 2}
	Sei a$\in$M, dann $a=\frac{-1}{n}$, n $\in\mathbb{N}$ $\frac{-1}{n1}\in$M\\
	$n+1>n\Rightarrow\frac{1}{n+1}<\frac{1}{n}\Rightarrow\frac{-1}{n+1}>\frac{-1}{n}\Rightarrow$M$\not\subseteq\frac{-1}{n}$ a 		keine obere Schranke$_{\square}$


\section{Satz}

	Sei K angeordneter Körper
		\begin{itemize}
			\item[1] Wenn M$\subset$K endlich und nicht leer, dann hat M ein Maximum und Minimum
			\item[2] (Wohlordnungsprinzip) Jede nichtleere Teilmenge M$\subset\mathbb{N}$ hat ein Minimum
		\end{itemize}

\subsection{Beweis}
	\begin{itemize}
		\item[zu 1] Klar.
		\item[zu 2] M ist nicht leer.\\
								Wähle n$\in$M $\{1,2,\cdots ,n\}\cap$M endlich und nicht leer, dann min($\{1,2,\cdots 													    					,n\}\cap$M)=min(M)$_{\square}$
	\end{itemize}


\section{Vollständigkeit - Definition}
	
	Sei K angeordneter Körper, M$\subseteq$K a$\in$K aheißt \underline{kleinste obere Schranke oder Supremum} von M wenn:
		\begin{itemize}
			\item[1] M$\leq$a und
			\item[2] kein b$\in$K mit b$<$a M$\leq$b erfüllt
		\end{itemize}
	a heißt \underline{größte untere Schranke oder Infimum} von M wenn:
		\begin{itemize}
			\item[1] a$\leq$a und
			\item[2] Kein b$\in$K mit a$<$b b$\leq$M erfüllt.
			\item[Notation] a=sup(M) bzw. a=inf(M) 
		\end{itemize}
		
\subsection{Bemerkung}
	Wenn a=max(M) dann a=sup(M)

\subsection{Beweis}
	Sei a= max(M)$\Rightarrow$M$\leq$a\\
	Seib$<$a	\underline{a$\in$M}, a$\not\leq$b$\Rightarrow$ M $\not\leq$ b $\Rightarrow$ a=sup(M)$_{\square}$	

\subsection{Bemerkung}
	Wenn ein Supremum existiert, dann ist es eindeutig
	
\subsection{Beweis}
	Angenommen a,b sind Suprema von M$\Rightarrow$M$\leq$a , M$\leq$b$\Rightarrow$a$\leq$b , b$\leq$a $\Rightarrow$ a=b
	
\subsection{Beispiel}
	$sup(\{\frac{-1}{n}|n\in\mathbb{N}\})=0$
